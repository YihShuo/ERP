2018/7/21 下午 03:37:51 select KCRKS.RKNO, KCRKS.CLBH, CLZL.YWPM , KCZLS.KCQty, KCZLS.KCBH,
       cast((select B.DDBH+',' from KCRKS A,DDZL B where B.DDBH=A.CGBH and A.RKNO=KCRKS.RKNO and A.CLBH=KCRKS.CLBH group by B.DDBH for XML Path ('') ) as varchar(max)) as DDBH, 
       cast((select B.Article+',' from KCRKS A,DDZL B where B.DDBH=A.CGBH and A.RKNO=KCRKS.RKNO and A.CLBH=KCRKS.CLBH group by B.Article for XML Path ('') ) as varchar(max)) as Article,

left join KCZLS on KCRK.CKBH=KCZLS.CKBH and KCRKS.CLBH=KCZLS.CLBH 
left join DDZL on KCRKS.CGBH=DDZL.DDBH 
where KCRKS.RKNO='20180704582' and KCRKS.CLBH='J010015889'
